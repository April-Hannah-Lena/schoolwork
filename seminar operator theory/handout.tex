\subsection{Lidskii's Theorem}

\begin{theorem}
    Let $X$ be a Hilbert space and $A \in L(X)$ be a trace-class operator with eigenvalues $\left\{ \lambda_1,\, \lambda_2,\, \ldots \right\}$. Then 
    \begin{equation}
        \Tr A = \sum_{n=1}^{\infty} \lambda_n . 
    \end{equation}
\end{theorem}
